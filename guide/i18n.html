# Change Language in component

Vue.component("language-changer", {
  template: "<div><a v-on:click=\"changeLanguage('de')\">DE</a></div>",
  methods: {
    changeLanguage(lang) {
      this.$i18n.i18next.changeLanguage(lang);
    }
  }
});
